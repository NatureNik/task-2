let readline = require('readline'); let rl = readline.createInterface({ input: process.stdin, output: process.stdout }); rl.question("Введите число: ", function(number) { number = parseInt(number); if (number >= 5 && number <= 10) { console.log("Число больше 5 и меньше 10"); } else { console.log("Неизвестное число"); } rl.close(); });
